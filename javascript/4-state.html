找出在数组中出现超过一半数的元素

        var gifts=[1,2,3,2,2]
        var value=0
        var newArr=[]
        for(var m=0;m<gifts.length;m++){
            if(newArr.indexOf(gifts[m]===-1)){
                newArr.push(gifts[m])
            }
        }//拿到去重后的新数组
      for(var j=0;j<newArr.length;j++){
          var count=0
            for(var i=0;i<gifts.length;i++){
                if(newArr[j]==gifts[i]){
                    count=count+1
                }
            }//通过变量，拿到j在gifts数组中出现的次数
            if(count > gifts.length/2){
                value=newArr[j]
            }
      }console.log(value)
