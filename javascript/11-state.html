    console.log(event.keyCode)
})

keypress keydown keyup

window.onkeypress=function(e){
    if(event.keyCode===13){
     console.log("回车")
  }
}

原生对象.blur()让原生对象失去焦点

原生对象.focus()让原生对象获得焦点

setTimeout(function(){},1000)过1秒执行function里内容，只执行一次

clearTimeout()停止setTimeout,1秒前停

setTimeout(function(){but.onclick()},1000)

onchange事件 用于chexkbox和select

oninput事件 当域的内容发生变化时触发，用于表单元素（很少用）

window.onload=function(){}文档加载完之后执行

event事件

只有绑定了事件就会存在事件对象 事件函数内部的参数 就是事件对象（可省略）

event.clientX获取鼠标距离窗口的左边距

event.clientY获取鼠标距离窗口的上边距

event.offsetX获取鼠标相对本身（触发事件的标签）的x偏移位置

event.offsetY鼠标相对于本身的y的偏移位置

event.pageX鼠标相对于文档的横轴偏移位置

event.pageY鼠标相对于文档的y轴偏移位置

event.stopPropagation()阻止事件冒泡

window.event.cancelBubble=true ie的阻止冒泡

document.querySelector("div").onclick=function(event){
    console.log(event)
}

绑定事件

.one .on

$(".box").on("click",function(){})

document.querySelector(".box").addEventListener("click",function(){})

原生对象.addEventListener("click",function(){}) 绑定事件监听（写多个都会执行,监听同样事件没有冲突）

原生对象.addEventListener("click",function(){},true) 多个加了true，执行事件捕获。冒泡从小到大，捕获从大到小。

超出出现滚动条overflow:auto;

滚动条

document.body.scrollTop获取文档的滚动条距离顶部的距离

原生对象.offsetLeft 只能获取原生对象相对于文档的左偏移位置（属性）

原生对象.offsetTop 只能获得原生对象相对于文档的上偏移位置（属性）

for in 语句通常用来遍历(全部查找)对象 for(var i in arr){}

$("input:checkbox").prop("checked")
