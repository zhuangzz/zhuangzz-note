模仿animate

  <button>按钮🔘</button>

      var box=document.querySelector(".box")
      var ani=null//对象

        function animate(obj,styleCss,target,callBack){
            var start=parseInt(window.getComputedStyle(obj,null)[styleCss])

            clearInterval(ani)//每次点击的时候删除上一次的setInterval的命令
            ani=setInterval(function(){
                if(start<target){
                    start=start+2
                }else if(start>target){
                    start=start-2
                }
                obj.style[styleCss]=start+"px"
                if(start==target){
                    clearInterval(ani)
                    if(callBack){
                        callBack()//每次执行的时候都要执行一遍callBack(),第一次执行的时候传进去的是没有执行的callBack()函数
                    }
                }
            },10)
    }

        document.querySelector("button").onclick=function(){
            animate(box,"marginLeft",0,function(){
            animate(box,"marginLeft",300)
            })
        }
