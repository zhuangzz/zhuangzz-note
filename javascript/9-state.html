            var oldName=obj.className.trim()
            var oldArr=oldName.split(" ")

            var x=-1
            for(var i=0;i<oldName.length;i++){
                if(classname==oldName[i]){
                    x=i
                    break;
                }
            }
            if(x==-1){
                obj.className=oldName+" "+classname
            }//这个写在for外面，如果写在else，那for循环过程中一有不等情况就会执行，且一定有不等情况
