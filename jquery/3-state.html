$("ul li a").click(function(){
    event.preventDefault();
    var ind=$(this).parent().index()
    var num=(ind * -1146)+"px"
    $(".pic").css("margin-left",num)
})

event.preventDefault();阻止默认跳转

overflow:hidden;溢出隐藏／父级没有高时写上可使父级有高撑起父级的高

.active{
            display: none;
        }

$("ul li a").click(function(){
                var ind=$(this).parent().index()+1
                $("div").eq(ind).removeClass("active")
                $("div").eq(ind).siblings("div").addClass("active")
})

.eq(索引值)选择索引值相应的标签，其中的索引值可以用变量表示。

.addClass("active");添加class名字

removeClass("active");移除class名字
