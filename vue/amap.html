1. 高德地图

1.1 轨迹

备选方案两种：PathSimplifier (轨迹展示)、 AMap.Polyline 折线），考虑到功能需求选择第一种

相关代码：

AMapUI.load(['ui/misc/PathSimplifier'],function(PathSimplifier){
  if (!PathSimplifier.supportCanvas) {
      alert('当前环境不支持 Canvas！');
      return;
  }
  self.initUiPage(PathSimplifier)
})

initUiPage: function(PathSimplifier) {
  var pathSimplifierIns = new PathSimplifier({
       zIndex: 100,
       map: mapinstance, //所属的地图实例
       getPath: function(pathData, pathIndex) {
           //返回轨迹数据中的节点坐标信息，[AMap.LngLat, AMap.LngLat...] 或者 [[lng|number,lat|number],...]
           return pathData.path;
       },
       getHoverTitle: function(pathData, pathIndex, pointIndex) {
           //返回鼠标悬停时显示的信息
           if (pointIndex >= 0) {
               //鼠标悬停在某个轨迹节点上
               return pathData.name + '，点:' + pointIndex + '/' + pathData.path.length;
           }
           //鼠标悬停在节点之间的连线上
           return pathData.name + '，点数量' + pathData.path.length;
       },
       renderOptions: {
           //轨迹线的样式
           pathLineStyle: {
               strokeStyle: 'red',
               lineWidth: 4,
               dirArrowStyle: true
           }
       }
   });
   //构建轨迹
   pathSimplifierIns.setData([{
       name: '轨迹0',
       path: [[121.5389385, 31.21515044], [121.5389385, 31.29615044], [121.5273285, 31.21515044],[116.322379,39.970632],[116.00,39.00],[116.6,39.970632],[116.0,39.98]]
   }]);
   //创建一个巡航器
   var navg0 = pathSimplifierIns.createPathNavigator(0, //关联第1条轨迹
    {
      loop: true, //循环播放
      speed: 500000
    })
   navg0.start();
}
