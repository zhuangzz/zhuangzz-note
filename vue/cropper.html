图片选择、剪裁、上传

在 vue 中使用cropperjs 实现图片选择、裁剪、上传

cropperjs官方npm库介绍


cropperjs github

npm安装插件：

npm install crpperjs --save

代码：

<template>
    <div class="upload">
        <div class="select-cover__first_select_btn" @mouseover="handleSwitchImg" @mouseleave="handleLeaveSec">
            <i class="el-icon-plus"></i>
            <p>选择封面</p>
        </div>
        <div class="select-cover__preview" @mouseover="handleSwitchImg" @mouseleave="handleLeaveSec">
            <div class="picture" :style="'backgroundImage:url('+headerImage+')'"></div>
        </div>
        <div class="setting-group__select_img_btn" v-show="isSecBtn" @mouseover="handleSwitchImg" @mouseleave="handleLeaveSec">
            <!-- 点击选择图片 -->
            <input type="button" value="上传文件" name="" id="" @click="updata">
            <input v-if="isReUpload" type="file" style="display:none" @change="getFile" id="input-file">
            <!-- <label class="ui-upload" for="change">上传图片<input type="file" id="change" accept="originImgUrl" @onchange="change" style="display:none"/></label> -->
        </div>
        <div class="zoomPreviewWrapper" v-show="isSelect">
            <div class="zoomPreviewWrapper_small">
                <div class="zoomOutPreview">
                    <img id="image" :src="originImgUrl" alt="Picture">
                </div>
                <div class="zoomInPreview-wrap">
                    <p>图片预览</p>
                    <div class="zoomInPreview"></div>
                </div>

                <button type="button" id="enterSec" @click="enterCropper">确定</button>
                <i class="el-icon-close" id="cancelSec" @click="cancelCropper"></i>
            </div>
        </div>        
    </div>
</template>

<script>
import Cropper from "cropperjs"
export default {
    data () {
        return {
            isReUpload: true,
            dialogVisible: true,
            cropper: "",
            headerImage: '', //展示的截图完成之后的图片
            picValue: '',
            isSecBtn: false,
            originImgUrl: '',
            // originImgUrl: '/static/img/imgBg.2baf229.jpg',
            isSelect: false,
            imgCropperData: {
                accept: "image/gif, image/jpeg, image/png, image/jpg"
            },
            croppable: false
        }
    },
    mounted() {
        var self = this
        var image = document.getElementById("image")
        self.cropper = new Cropper(image, {
            aspectRatio: 654/358,//宽高比
            viewMode: 1,
            background: false,
            zoomable: true,
            preview: '.zoomInPreview',
            //dragCrop: false,//默认值true,是否允许移除当前的剪裁框
            // movable: false,//默认值true是否允许移动剪裁框
            //resizable: false,//默认值true,是否允许改变剪裁框的大小
            ready: function() {
                self.croppable = true
            }
        })     
    },
    methods: {
        updata () {
            console.log("点击上传")
            // this.isReUpload = true
            $('#input-file').click()
        },
        getFile (e) {
            let files = e.target.files || e.dataTransfer.files
            console.log("files",files)
            if (!files.length) return
            let type = files[0].type //文件的类型，判断是否是图片
            let size = files[0].size //文件的大小，判断图片的大小
            if (this.imgCropperData.accept.indexOf(type) == -1) {
                alert("请选择我们支持的图片格式！")
                return false
            }
            if (size > 5242880) {
                alert("请选择5M以内的图片！")
                return false
            }
            this.picValue = files[0]
            this.originImgUrl = this.getObjectURL(this.picValue)
            //每次替换图片要重新得到新的url
            if (this.cropper) {
                this.cropper.replace(this.originImgUrl)
            }
            this.isSelect = true
                this.doSomething(e)
        },
        doSomething(e){ // do something
            this.fileName = e.target.files[0].name
            this.isReUpload = false
            this.isReUpload = true
        },
        handleSwitchImg () {
            this.isSecBtn = true
        },
        handleLeaveSec () {
            this.isSecBtn = false
        },
        //获取到上传的文件
        referenceUpload(e) {
           let files = e.target.files || e.dataTransfer.files
           console.log("files",files)
           if (!files.length) return
           let type = files[0].type //文件的类型，判断是否是图片
           let size = files[0].size //文件的大小，判断图片的大小
           if (this.imgCropperData.accept.indexOf(type) == -1) {
               alert("请选择我们支持的图片格式！")
               return false
           }
           if (size > 5242880) {
               alert("请选择5M以内的图片！")
               return false
           }
           this.picValue = files[0]
           this.originImgUrl = this.getObjectURL(this.picValue)
           //每次替换图片要重新得到新的url
           if (this.cropper) {
               this.cropper.replace(this.originImgUrl)
           }
           this.isSelect = true
        },
        cancelCropper () {
            this.isSelect = false
            var obj = document.getElementById('change')
            // console.log("obj",obj)
            // obj.outerHTML=obj.outerHTML
        },
        //创建url路径
        getObjectURL(file) {
            var url = null
            if (window.createObjectURL != undefined) {
                // basic
                url = window.createObjectURL(file)
            } else if (window.URL != undefined) {
                // mozilla(firefox)
                url = window.URL.createObjectURL(file)
            } else if (window.webkitURL != undefined) {
                // webkit or chrome
                url = window.webkitURL.createObjectURL(file)
            }
            return url
        },
        //确定提交
        enterCropper() {
            this.isSelect = false
            var croppedCanvas
            var roundedCanvas
            if (!this.croppable) {
                return
            }
            // Crop
            croppedCanvas = this.cropper.getCroppedCanvas()
            // Round
            roundedCanvas = this.getRoundedCanvas(croppedCanvas)
            this.headerImage = roundedCanvas.toDataURL()
            //上传图片

            var cas = $('#image').Cropper('getCroppedCanvas')//获取被裁剪后的canvas
            var base64url = cas.toDataURL('image/png') //转换为base64地址形式

            console.log("base64url",base64url)
            alert("上传成功")
        },
        //canvas画图
        getRoundedCanvas(sourceCanvas) {
            var canvas = document.createElement("canvas")
            var context = canvas.getContext("2d")
            var width = sourceCanvas.width
            var height = sourceCanvas.height
            canvas.width = width
            canvas.height = height
            context.imageSmoothingEnabled = true
            context.drawImage(sourceCanvas, 0, 0, width, height)
            context.globalCompositeOperation = "destination-in"
            context.beginPath()
            // context.arc(
            //   width / 2,
            //   height / 2,
            //   Math.min(width, height) / 2,
            //   0,
            //   2 * Math.PI,
            //   true
            // )
            context.fill()

            return canvas
        },    
    }
}
</script>
