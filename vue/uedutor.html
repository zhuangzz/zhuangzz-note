<script>
  import '../../static/ue/ueditor.config.js'
  import '../../static/ue/ueditor.all.min.js'
  import '../../static/ue/lang/zh-cn/zh-cn.js'
  import '../../static/ue/ueditor.parse.min.js'
  import '../../static/ue/jquery-3.3.1.min.js'

  export default {
    name: 'UE',
    data () {
      return {
        editor: null
      }
    },
    props: {
      defaultMsg: {
        type: String
      },
      config: {
        type: Object
      }
    },
    mounted() {
      const _this = this;
      this.editor = UE.getEditor('editor', this.config); // 初始化UE
      this.editor.addListener("ready", function () {
        // _this.editor.setContent(_this.defaultMsg); // 确保UE加载完成后，放入内容。
        _this.editor.setHeight(500);
        _this.setUEContent(_this.defaultMsg)
      });
    },
    methods: {
      // 获取内容方法
      getUEContent() { 
        return this.editor.getContent()
      },
      // 获取纯文本
      getUEContentText () {
        // return this.editor.getContentTxt() // 纯文本
        return this.editor.getPlainTxt() // 带格式的纯文本
      },
      setUEContent(defaultMsg) {
        const _this = this;
        _this.editor.setContent(defaultMsg); // 确保UE加载完成后，放入内容。
      }
    },
    destroyed() {
      this.editor.destroy();
    }
  }
</script>
